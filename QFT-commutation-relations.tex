	\begin{align*}
		\phi(x) &= \phi^{+}(x) + \phi^{-}(x) \,,\\
		\phi^{+}(x) &= \int \frac{\dd^3 k}{(2\pi)^{3/2}}
			\frac{1}{\sqrt{2 \omega_{\mathbf{k}}}} \,
			a(\mathbf{k}) \ee^{-\ii k\cdot x}
			\,,\\
		\phi^{-}(x) &= \int \frac{\dd^3 k}{(2\pi)^{3/2}}
			\frac{1}{\sqrt{2 \omega_{\mathbf{k}}}} \,
			a^\dagger(\mathbf{k}) \ee^{\ii k\cdot x}\,.
	\end{align*}
	Contour integrals: \vspace{-2mm}
	\begin{gather*}
	\oint_\gamma \frac{f(z) \,\dd z}{z-z_0}= 
		\begin{cases} 
			\, 2\pi\ii \,f(z_0), & z_0 \in \op{int}\gamma , \\
			\, 0, & z_0 \notin \op{int}\gamma ;
		\end{cases} \\[1.5mm]
	I_\gamma(x) = \oint_\gamma \frac{\ee^{-\ii k\cdot x}\,\dd k^0}{k^2-m^2}
	= \oint_\gamma \frac{\ee^{-\ii k\cdot x}\,\dd k^0}{(k^0-\omega_{\mathbf{k}})(k^0+\omega_{\mathbf{k}})} \,.
	\end{gather*}
	Positive and negative frequency poles: \vspace{-3mm}
	\begin{align*}
	I_\gamma(x) &= 
			\, + 2\pi\ii \, \frac{\ee^{-\ii k\cdot x}}{2\omega_{\mathbf{k}}},
			\quad k^0 = + \omega_{\mathbf{k}} \in \gamma=C^{+} , \\
	I_\gamma(x) &= 
			\, - 2\pi\ii \, \frac{\ee^{-\ii k\cdot x}}{2\omega_{\mathbf{k}}},
			\quad k^0 = - \omega_{\mathbf{k}} \in \gamma=C^{-} , \\
	I_{\pm}(x) &= 
			\, \pm 2\pi\ii \, \frac{\ee^{-\ii k\cdot x}}{2\omega_{\mathbf{k}}}
			\delta(k^0\mp\omega_{\mathbf{k}}) .
	\end{align*}
	Hence,\vspace{-3mm}
	\begin{gather*}
	\Delta_\gamma(x) = \int_\gamma \! \frac{\dd^4 k}{(2\pi)^4}
	\ee^{-\ii k\cdot x} \frac{-1}{k^2-m^2}	= 
		-\int\! \frac{\dd^3 k}{(2\pi)^4} I_\gamma \,, \\
	\ii \Delta^{\pm}(x) = 
		\pm \int \frac{\dd^3 k}{(2\pi)^3} 
		\ee^{-\ii k\cdot x} \, \delta(k^0\mp\omega_{\mathbf{k}})\frac{1}{2\omega_{\mathbf{k}}} ,
	\end{gather*} \\[-2mm]

	\begin{gather*}
	\delta(k^2-m^2) \, \theta(\pm k^0) = 
		\delta(k^0\mp\omega_{\mathbf{k}}) \frac{1}{2\omega_{\mathbf{k}}} \,,\\
	\pm \int\dd k^0 \, \delta(k^2-m^2) \, \theta(\pm k^0) = 
		\frac{1}{2\omega_{\mathbf{k}}} \,.
	\end{gather*}

	$1^\circ$ The commutation functions are Lorentz invariant.\vspace{-1mm}
	\begin{align}
	\ii\Delta(x) &= \int\! \frac{\dd^3 k}{(2\pi)^3}
		\int \!\dd k^0 \, \ee^{-\ii k\cdot x}
		\delta(k^2-m^2) \, \op{sgn}(k^0) \\
	\ii\Delta_1(x) &= \int\! \frac{\dd^3 k}{(2\pi)^3}
		\int\! \dd k^0 \, \ee^{-\ii k\cdot x}
		\delta(k^2-m^2) \\
	\ii \Delta^{\pm}(x) &= \int\! \frac{\dd^3 k}{(2\pi)^3}
		\int\! \dd k^0 \,\ee^{-\ii k\cdot x} 
		\delta(k^2-m^2) \, \left(\pm \theta(\pm k^0)\right)
	\end{align}
	$2^\circ$  The following \textit{initial conditions at vanishing time difference} hold:\vspace{-3mm}
	\begin{equation}
	\Delta(0,\mathbf{x}) = 0, \qquad
	\partial_0 \Delta(x^0,\mathbf{x}) \vert_{x^0=0} = -\delta^3(\mathbf{x})\,.
	\vspace{-3mm}
	\end{equation}
	As a corrolary, the ETCR are regained.\\[2mm]
	$3^\circ$  From $2^\circ$, it also follows the condition of \textit{microcausality}:
	$\Delta(x-y)=0$ for $(x-y)^2<0$,
	i.e. $\Delta(x-y)$ vanishes if the argument is spacelike.
	(This is a very fundamental property!)
	\\[2mm]
	$4^\circ$  All the commutation functions satisfy EoM; 
	e.g. $ - ( \square_x + m^ 2) \, \Delta(x-y) = 0 $.
	\\[2mm]
	Note that $2^\circ$ and $4^\circ$ are sufficient to provide a unique 
	definition of the function $\Delta(x)$.
	(EoM is a hyperbolic PDE with the initial data (4) on a space-like 
	hypersurface $x^0 =0$).
	\\[1mm]
	Also, all the $\Delta$-functions show singular behavior on the light cone.
};

	\begin{align*}
	\Delta^{\pm}(-x) &= -\Delta^{\mp}(x) \\[1mm]
	\Delta(-x) &= -\Delta(x) \\[1mm]

	\Delta(x) &= \Delta^{+}(x) + \Delta^{-}(x) \\[1mm]
	\Delta_1(x) &= \Delta^{+}(x) - \Delta^{-}(x) \\[1mm]
	\Delta^{+}(x) &= \tfrac{1}{2} \left( \Delta(x) + \Delta_1(x) \right) \\[1mm]
	\Delta^{-}(x) &= \tfrac{1}{2} \left( \Delta(x) - \Delta_1(x) \right) \\[1mm]
	\Delta_1(x) &= \Delta_{\mathrm{F}}(x) - \Delta_{\mathrm{D}}(x) \\[1mm]
	\Delta(x) &= \Delta_{\mathrm{R}}(x) - \Delta_{\mathrm{A}}(x)
	\end{align*}

	Namely, they contain a product of the function 
	$\Delta(x)$ with a unit step function in time,
	either $\theta(x^0)$ or $\frac{1}{2}\op{sgn}(x^0)$, of which
	derivativs are the $\delta$-functions.\vspace{2mm}
	\\For example, for
	$(\square+m^2)\Delta_{\mathrm{F}}(x)$ we have:\vspace{-1mm}
	\begin{gather*}
	(\square+m^2) \left( \tfrac{1}{2} \left( \op{sgn}(x^0) \, \Delta(x) + \Delta_1(x) \right) \right) 
	(\delta_0^2 -\boldsymbol{\nabla}^2+m^2) \left( \tfrac{1}{2} \op{sgn}(x_0) \Delta(x) \right)= \\
	= \left( \partial_0 \delta(x) \right)\Delta(x)
	+ 2 \delta(x^0) \left( \partial_0 \Delta(x) \right)
	+ \tfrac{1}{2} \op{sgn}(x_0) (\square+m^2)\Delta(x) \,.
	\\[-7mm]
	\end{gather*}The last term vanishes because $\Delta(x)$ solves EoM,
	and the first term is equivalent to $-\delta(x^0)\!\left( \partial_0\Delta(x) \right)$. 
	Using a test function $f(x^0)$, an integration by parts yields,\vspace{-2mm}
	\begin{gather*}
	\int \dd x^0 \left( \partial_0 \delta(x) \right)\Delta(x) f(x^0)
	= - \partial_0\Delta(x) \,\vert_{x^0=0} \,f - \Delta(0,\mathbf{x})\, \partial_0 \,f \,\vert_{x^0=0}
	= - \partial_0 \Delta(x) \,\vert_{x^0=0} \,f = -\delta^3(\mathbf{x}) \,f \,.
	\\[-8mm]
	\end{gather*}
	Here, we applied the initial conditions (4) from Property $2^\circ$. Hence,\vspace{-1mm}
	\begin{gather*}
	(\square+m^2)\Delta_{\mathrm{F}}(x) 
	= - \delta(x^0) \left( \partial_0 \Delta(x) \right)
	+ 2 \delta(x^0) \left( \partial_0 \Delta(x) \right)
	= \delta(x^0) \partial_0 \Delta(x) = -\delta(x^0) \delta^3(\mathbf{x}) 
	= -\delta^4(x) \,.
	\end{gather*}
};

\node [boxRed,R=15,C=80,align=left] at (3,11.5) {
	\textsf{\textbf{ Feynman propagator}\\
	(causal propagator or causal Green's function)}
};

\node [boxYellow,R=40,C=80,align=center] at (3,10) {\vspace{-5mm}
	\begin{align*}
	\Delta_{\mathrm{F}}(x-y) &= \int_{C_{\mathrm{F}}} \frac{\dd^4 k}{(2\pi)^4}
	   \ee^{-\ii k \cdot (x-y)}
	   \frac{1}{k^2-m^2} \\[2mm]
	\ii \Delta_{\mathrm{F}}(x-y) &= \left\langle 0 \vert \mathrm{T} \{ \, \phi(x), \, \phi(y) \, \} \vert 0 \right\rangle \\[2mm]
	\Delta_{\mathrm{F}}(x) &= \theta(x^0) \, \Delta^{+}(x) - \theta(-x^0) \, \Delta^{-}(x) \\[2mm]
	 &= \tfrac{1}{2} \left( \op{sgn}(x^0) \, \Delta(x) + \Delta_1(x) \right)
	\end{align*}
